# Базовые команды Unix

## Информация о системе

``` bash
# Текущий пользователь
whoami

# UID, GID и группы пользователя
id

# Информация о ядре
uname -a

# Информация о дистрибутиве Linux
cat /etc/os-release

# Информация о версии macOS
sw_vers
```

---

## Некоторые базовые утилиты

``` bash
# Создание папки
mkdir <name>

# Переход в указанный каталог
cd /path/

# Создание файла
touch file.txt

# Запись текста в файл
echo "Hello" > file.txt

# Вывод содержимого файла на экран
cat file.txt

# Поиск строк по шаблону в тексте
grep "text" file.txt

# Переход в домашний каталог
cd ~

# Вывод списка файлов и каталогов
ls

# Подробный список (права, владелец, размер)
ls -l

# Показ скрытых файлов
ls -a

# Очистка экрана терминала
clear
```

---

## Процесс как абстракция

``` bash
# Запуск процесса
sleep 100

# Текущие процессы терминала
ps

# Все процессы системы
ps aux

# Динамический мониторинг
top
```

---

## Идентификаторы процессов

``` bash
# Поиск PID по имени процесса
pgrep sleep
```

---

## Завершение процесса

``` bash
# Завершение процесса
kill <PID>

# Принудительное завершение
kill -9 <PID>
```

---

## Переменные окружения

``` bash
# Список переменных окружения
env

# Путь к домашнему каталогу текущего пользователя
echo $HOME

# Список каталогов, в которых оболочка ищет исполняемые файлы при вводе команды
echo $PATH
```
